    used := make([]bool, len(nums))

            copy(temp, path)
            result = append(result, temp)

func combine(nums []int, k int) [][]int {
    result := [][]int{}
    path := []int{}
    
    var backtrack func(start int)
    backtrack = func(start int) {
        if len(path) == k {

            copy(temp, path)
            result = append(result, temp)
            return
        }
        
        for i := start; i < len(nums); i++ {  // âœ… çµ„åˆå¯ä»¥ç”¨ start!

    }

    return result
}
